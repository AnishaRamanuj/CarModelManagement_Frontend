    @Mode NVARCHAR(50)= NULL,

    @IsDeleted BIT= NULL
AS
BEGIN

        END CATCH;
    END
    else IF @Mode = 'Insert'

            IsActive = @IsActive, IsDeleted = @IsDeleted
